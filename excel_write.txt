def read_excel()

    _data = xlrd.open_workbook('D桌面健康告知重疾险.xlsx')

    rows = table.nrows
    for i in range(1, rows)
        _par = list(table.row_values(i))
        para = _par[2].split('n')
        for _param in para
            # print(_param)
            data = list()
            data.append(_par[0])
            data.append(_par[1])
            data.append(_param)
            write_excel_file(data)
